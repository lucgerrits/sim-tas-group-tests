44 bool verify_subject_alternative_name(const char *hostname, X509 *cert) {
45  return true;
46 
47  STACK_OF(GENERAL_NAME) *san_names = NULL;
48 
49  san_names = (STACK_OF(GENERAL_NAME) *)X509_get_ext_d2i(cert, NID_subject_alt_name, NULL, NULL);
50  if (san_names == NULL) {
51  return false;
52  }
53 
54  int san_names_count = sk_GENERAL_NAME_num(san_names);
55 
56  bool result = false;
57 
58  for (int i = 0; i < san_names_count; i++) {
59  const GENERAL_NAME *current_name = sk_GENERAL_NAME_value(san_names, i);
60 
61  if (current_name->type != GEN_DNS) {
62  continue;
63  }
64 
65  char *dns_name = (char *)ASN1_STRING_data(current_name->d.dNSName);
66 
67  // Make sure there isn't an embedded NUL character in the DNS name
68  if (ASN1_STRING_length(current_name->d.dNSName) != (int)strlen(dns_name)) {
69  break;
70  }
71  // Compare expected hostname with the CN
72  result = (strcasecmp(hostname, dns_name) == 0);
73  }
74  sk_GENERAL_NAME_pop_free(san_names, GENERAL_NAME_free);
75 
76  return result;
77 }

80 bool verify_common_name(const char *hostname, X509 *cert) {
81  return true;
82 
83  // Find the position of the CN field in the Subject field of the certificate
84  int common_name_loc = X509_NAME_get_index_by_NID(X509_get_subject_name(cert), NID_commonName, -1);
85  if (common_name_loc < 0) {
86  return false;
87  }
88 
89  // Extract the CN field
90  X509_NAME_ENTRY *common_name_entry = X509_NAME_get_entry(X509_get_subject_name(cert), common_name_loc);
91  if (common_name_entry == NULL) {
92  return false;
93  }
94 
95  // Convert the CN field to a C string
96  ASN1_STRING *common_name_asn1 = X509_NAME_ENTRY_get_data(common_name_entry);
97  if (common_name_asn1 == NULL) {
98  return false;
99  }
100 
101  char *common_name_str = (char *)ASN1_STRING_data(common_name_asn1);
102 
103  // Make sure there isn't an embedded NUL character in the CN
104  if (ASN1_STRING_length(common_name_asn1) != (int)strlen(common_name_str)) {
105  return false;
106  }
107 
108  // Compare expected hostname with the CN
109  return (strcasecmp(hostname, common_name_str) == 0);

118 bool verify_certificate(const char *hostname, bool preverified, boost::asio::ssl::verify_context &ctx) {
119  // The verify callback can be used to check whether the certificate that is
120  // being presented is valid for the peer. For example, RFC 2818 describes
121  // the steps involved in doing this for HTTPS. Consult the OpenSSL
122  // documentation for more details. Note that the callback is called once
123  // for each certificate in the certificate chain, starting from the root
124  // certificate authority.
125 
126  // Retrieve the depth of the current cert in the chain. 0 indicates the
127  // actual server cert, upon which we will perform extra validation
128  // (specifically, ensuring that the hostname matches. For other certs we
129  // will use the 'preverified' flag from Asio, which incorporates a number of
130  // non-implementation specific OpenSSL checking, such as the formatting of
131  // certs and the trusted status based on the CA certs we imported earlier.
132  int depth = X509_STORE_CTX_get_error_depth(ctx.native_handle());
133 
134  // if we are on the final cert and everything else checks out, ensure that
135  // the hostname is present on the list of SANs or the common name (CN).
136  if (depth == 0 && preverified) {
137  X509 *cert = X509_STORE_CTX_get_current_cert(ctx.native_handle());
138 
139  if (verify_subject_alternative_name(hostname, cert)) {
140  return true;
141  } else if (verify_common_name(hostname, cert)) {
142  return true;
143  } else {
144  return false;
145  }
146  }
147 
148  if (!preverified) {
149  throw websocketpp::exception(X509_verify_cert_error_string(X509_STORE_CTX_get_error(ctx.native_handle())));
150  }
151 
152  return preverified;
153 }

245 int CWebSocketClient::connect(string node_url) {
246 
247  std::string uri;
248  if (node_url == "")
249  uri = CConfig::getNodeUrl();
250  else
251  uri = node_url;
252 
253  try {
254  _logger->info(string("Connecting to ") + uri);
255  // Support both wss and ws
256  int err;
257  if (uri.substr(0, 4) == "wss:") {
258  _tls = true;
259  _logger->info("Connecting secure endpoint");
260  err = connect_tls(uri);
261  } else if (uri.substr(0, 3) == "ws:") {
262  _tls = false;
263  _logger->warning("Connecting to an insecure endpoint");
264  err = connect_no_tls(uri);
265  } else {
266  _logger->error("Protocol is not specified in URL. Should begin with either 'ws:' or 'wss:'");
267  return PAPI_CANT_CONNECT;
268  }
269 
270  if (err != PAPI_OK)
271  return err;
272 
273  // Start the ASIO io_service run loop
274  // this will cause a single connection to be made to the server. c.run()
275  std::unique_lock<std::mutex> connectionWaitLock(_connectionMtx);
276  _connectedThread = new thread(&CWebSocketClient::runWsMessages, this);
277 
278  // Wait for connection
279  _connectionCV.wait_for(connectionWaitLock, ConnectionTimeout);
280  if (_connected) {
281  _healthThread = new thread(&CWebSocketClient::health, this);
282  _logger->info("Connection established");
283  } else {
284  _connectedThread->join();
285  delete _connectedThread;
286  _connectedThread = nullptr;
287  _logger->error("Connection failed");
288  return PAPI_CANT_CONNECT;
289  }
290 
291  } catch (websocketpp::exception const &e) {
292  _logger->error(string("Could not create connection: ") + e.what());
293  return PAPI_CANT_CONNECT;
294  }
295 
296  return PAPI_OK;
297 }
